  <div class="max-w-2xl mx-auto py-8">

    <h1 class="text-2xl font-bold mb-6 text-green-900 text-center">Inventário de Carros</h1>

    <form id="carForm" class="bg-white p-4 rounded shadow mb-6 flex flex-col gap-2">

      <input type="text" id="name" placeholder="Nome do carro" class="border rounded px-2 py-1" required>

      <input type="number" id="year" placeholder="Ano" class="border rounded px-2 py-1" required>

      <input type="number" id="price" placeholder="Preço" class="border rounded px-2 py-1" required>

      <input type="text" id="image" placeholder="URL da imagem" class="border rounded px-2 py-1">

      <button type="submit" class="bg-green-700 text-white py-2 rounded hover:bg-green-800">Adicionar Carro</button>

    </form>

    <ul id="carList" class="space-y-4"></ul>

  </div>

  <script>

    async function loadCars() {

      const res = await fetch('/api/cars');

      const cars = await res.json();

      const carList = document.getElementById('carList');

      carList.innerHTML = '';

      cars.forEach(car => {

        const li = document.createElement('li');

        li.className = 'bg-white p-4 rounded shadow flex justify-between items-center';

        li.innerHTML = `

          <div>

            <strong>${car.name}</strong> (${car.year})<br>

            <span class="text-green-700 font-bold">R$ ${car.price}</span>

          </div>

          <button onclick="deleteCar(${car.id})" class="bg-red-600 text-white px-3 py-1 rounded hover:bg-red-800">Remover</button>

        `;

        carList.appendChild(li);

      });

    }

    async function deleteCar(id) {

      if (confirm('Tem certeza que deseja remover este carro?')) {

        await fetch('/api/admin/cars/' + id, { method: 'DELETE' });

        loadCars();

      }

    }

    document.getElementById('carForm').onsubmit = async function(e) {

      e.preventDefault();

      const car = {

        name: document.getElementById('name').value,

        year: parseInt(document.getElementById('year').value),

        price: parseFloat(document.getElementById('price').value),

        image: document.getElementById('image').value

      };

      await fetch('/api/admin/cars', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(car)

      });

      e.target.reset();

      loadCars();

    }

    loadCars();

  </script>
